#version 400

// Input Variables
in vec3 normal;
in vec3 lightDir;

// Uniform Variables
uniform vec4 diffuseMaterialColor;

// Output Variables
out vec4 fragColor;

void main() 
{
	const vec4 AmbientColor = vec4(0.2, 0.0, 0.0, 1.0);

	vec3 lightV = normalize(lightDir);
	float diffuseDot = dot(normal, lightV);
	float diffuseTerm = clamp(diffuseDot, 0.0, 1.0);

    fragColor = AmbientColor + diffuseMaterialColor * diffuseTerm;
}
